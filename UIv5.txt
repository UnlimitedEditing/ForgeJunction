tsexport interface Lora {
  id: string
  name: string
  slug: string
  description?: string
  thumbnail_url?: string | null
  compatible_workflows?: string[]  // workflow slugs this LoRA works with
  trigger_words?: string[]
}

export async function fetchLoras(workflowSlug?: string): Promise<Lora[]> {
  // Try endpoint: ${BASE_URL}loras/ or ${BASE_URL}concepts/
  // If the endpoint 404s, return [] and log a warning rather than throwing.
  // Pass workflowSlug as a query param if provided: ?workflow=<slug>
  const url = workflowSlug
    ? `${BASE_URL}loras/?workflow=${workflowSlug}`
    : `${BASE_URL}loras/`
  try {
    const res = await fetch(url, { headers: headers() })
    if (!res.ok) return []
    const json = await res.json()
    return (json.data ?? json).map((item: { id: string; attributes?: Lora }) => ({
      id: item.id,
      ...(item.attributes ?? item),
    }))
  } catch {
    return []
  }
}
